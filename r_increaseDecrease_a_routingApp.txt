1) Build a React component with two buttons (+ and -) to increase or decrease a numeric value. - 10M

npx create-react-app count
cd count
npm start

 
App.js


import React, { useState } from "react";
function App() {
  const[count,setCount]=useState(0);
return (
<div>
<h1>Counter : {count}</h1>
<button onClick={()=>setCount(count+1)} style={{ fontSize: "20px", padding: "10px 20px" }}> + </button>
<button onClick={()=>setCount(count-1)} style={{ fontSize: "20px", padding: "10px 20px" , marginLeft:"10px"}}> - </button>
</div>
);
}
export default App;



2) Design a Routing Demo app in Angular with Home, About, and Contact Components 15M


ng new routing  - - routing
cd routing
ng generate component home 
ng generate component about
ng generate component contact 
ng serve 



app.html
<h1>Simple Angular Routing App</h1>
<nav>
  <a routerLink="">Home</a> |
  <a routerLink="/about" routerLinkActive="active">About</a> | 
   <a routerLink="/contact" routerLinkActive="active">Contact</a>
</nav>
<hr />
<router-outlet></router-outlet>



app.routes.ts
import { Routes } from '@angular/router';
import { Home } from './home/home';
import { About } from './about/about';
import { Contact } from './contact/contact';
export const routes: Routes = [
 { path: '', component: Home },        // default path
 { path: 'about', component: About },
 { path: 'contact', component: Contact },  // /about path
];



app.ts
import { Component, signal } from '@angular/core';
import { RouterOutlet } from '@angular/router';
import { RouterModule } from '@angular/router';
@Component({
  selector: 'app-root',
  standalone: true,
  imports: [RouterOutlet, RouterModule],
  templateUrl: './app.html',
  styleUrl: './app.css'
})
export class App {
  protected readonly title = signal('routing-HAC');
}
