1.Build a React app with a colored box that changes to a random color when clicked.

npx create-react-app color_demo
cd color_demo
npm start


App.js


import React, { useState } from 'react';
function App() {
  const [color, setColor] = useState('lightblue');

  function changeColor() {
    const random = '#' + Math.floor(Math.random()*16777215).toString(16);
    setColor(random);
  }
return (
    <div
      onClick={changeColor}
      style={{ width: 200, height: 200, background: color, margin: '50px auto' }}>
    </div>
  );
}
export default App;
