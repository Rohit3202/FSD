1) Design a React app that displays the current date and time when a button is clicked -10 M

npx create-react-app time
cd time
npm start 


App.js



 import React , {useState} from "react";
function App() {
    const[time,setTime] = useState("");
  return (
    <div style={{textAlign : "center" , marginTop : 80 }}>
        <p>{time}</p>
    <button onClick={()=> setTime(new Date().toLocaleString())}> Show Date and Time </button>
    </div>
      );
}
export default App;
