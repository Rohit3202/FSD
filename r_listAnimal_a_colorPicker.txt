Q1.Create a React app with a dropdown list of animals. When the user selects one, display the selected animal name below


npx create-react-app animaldrop
cd animaldrop
npm start


1)App.js


import React, { useState } from "react";
function App() {
  const [selectedAnimal, setSelectedAnimal] = useState("");
  const animals = ["Dog", "Cat", "Elephant", "Tiger", "Lion"];
  return (
    <div style={{ textAlign: "center", marginTop: "50px" }}>
      <h3>Select an Animal</h3>
      <select onChange={(e) => setSelectedAnimal(e.target.value)}>
        <option value="">--Choose--</option>
        {animals.map((animal, i) => (
          <option key={i} value={animal}>
            {animal}
          </option>
        ))}
      </select>
      {selectedAnimal && <h4>You selected: {selectedAnimal}</h4>}
    </div>
  );
}
export default App;


Q2.Build a Color Picker app in Angular where the user selects a color and applies it to a div box

ng new color
cd color
ng serve


1)App.component.ts


import { Component } from '@angular/core';
import { RouterOutlet } from '@angular/router';
@Component({
  selector: 'app-root',
  imports: [RouterOutlet],
  templateUrl: './app.component.html',
  styleUrl: './app.component.css'
})
export class AppComponent {
  selectedColor = '#ffffff';
  changeColor(event: any) {
    this.selectedColor = event.target.value;
  }
}



2)App.component.html



<h3>Simple Color Picker</h3>
<input type="color" (input)="changeColor($event)">
<div 
  [style.background-color]="selectedColor"
  style="width:200px; height:200px; border:1px solid black; margin-top:10px;">
</div>
