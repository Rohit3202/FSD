1) Develop a React app that generates and displays a random number every time a button is Pressed -  (10M)                                                                                                                                           
npx create-react-app random 
cd random
npm start 



App.js


import { useState } from "react";
function App() {
  const [n, setN] = useState(0);
  return (
    <div style={{ textAlign: "center", marginTop: 50 }}>
      <h2>{n}</h2>
      <button onClick={() => setN(Math.floor(Math.random() * 100))}>
        Generate
      </button>
    </div>
  );
}
export default App;
